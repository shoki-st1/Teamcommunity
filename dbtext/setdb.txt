--sqlの設定--
--dbの作成--
CREATE DATABASE IF NOT EXISTS userdb;
USE userdb;

--usertableの作成--
CREATE TABLE IF NOT EXISTS userdb_table
(
  userid int(15) NOT NULL AUTO_INCREMENT,
  password VARCHAR(255) NOT NULL,
  PRIMARY KEY (`userid`)
  );

--task_tableの作成--
CREATE TABLE IF NOT EXISTS task_table
(
  userid VARCHAR(50),
  task VARCHAR(100),
  startday VARCHAR(25),
  finday VARCHAR(25)
);
